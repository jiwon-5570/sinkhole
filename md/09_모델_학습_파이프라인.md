# 국토안전관리 프로젝트 모델 학습 파이프라인

## 1. 문서 목적

이 문서는 국토안전관리 프로젝트에서 사용할 모델 학습 흐름을 정의한다.

본 가이드는 다음을 목표로 한다.

1. 데이터 → feature → 모델 → 결과까지 전체 흐름을 연결한다.
2. 실제 코드로 바로 구현 가능한 구조를 제공한다.
3. 규칙 기반 + 머신러닝 결합 구조를 명확히 한다.
4. 학습/평가/저장/서비스 연결까지 포함한다.

4. 단계별 상세 파이프라인
4.1 데이터 로딩
목적
raw 데이터 불러오기
코드 예시
import pandas as pd

df_sinkhole = pd.read_csv("data/raw/sinkhole.csv")
df_weather = pd.read_csv("data/raw/weather.csv")
df_gpr = pd.read_csv("data/raw/gpr.csv")
4.2 데이터 정제
목적
결측값 처리
이상값 제거
데이터 형식 통일
코드 예시
df_sinkhole = df_sinkhole.dropna(subset=["region_id"])
df_weather["rainfall"] = df_weather["rainfall"].fillna(0)
4.3 Feature 생성
목적
모델 입력 변수 생성
코드 예시
feature_df = pd.DataFrame()

feature_df["region_id"] = df_sinkhole["region_id"]

feature_df["past_sinkhole_count"] = df_sinkhole.groupby("region_id").size()

feature_df["rainfall_7d"] = df_weather.groupby("region_id")["rainfall"].rolling(7).sum().values

feature_df["gpr_detected_count"] = df_gpr.groupby("region_id")["cavity_detected"].sum()
4.4 Feature Dataset 생성
목적
모델 입력용 데이터 완성
feature_df = feature_df.fillna(0)
feature_df.to_csv("data/feature_dataset.csv", index=False)
4.5 Target 생성
목적
학습용 정답 데이터 생성
방법 1: 규칙 기반 점수
feature_df["risk_score"] = (
    feature_df["past_sinkhole_count"] * 2 +
    feature_df["gpr_detected_count"] * 3 +
    feature_df["rainfall_7d"] * 0.5
)
방법 2: 등급 변환
def classify_risk(score):
    if score < 30:
        return 0
    elif score < 60:
        return 1
    elif score < 80:
        return 2
    else:
        return 3

feature_df["risk_level"] = feature_df["risk_score"].apply(classify_risk)
4.6 Train / Test 분리
from sklearn.model_selection import train_test_split

X = feature_df.drop(["risk_score", "risk_level"], axis=1)
y = feature_df["risk_score"]

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
4.7 모델 학습 (XGBoost)
import xgboost as xgb

model = xgb.XGBRegressor(
    n_estimators=100,
    max_depth=6,
    learning_rate=0.1
)

model.fit(X_train, y_train)
4.8 모델 평가
from sklearn.metrics import mean_absolute_error

y_pred = model.predict(X_test)

mae = mean_absolute_error(y_test, y_pred)
print("MAE:", mae)
4.9 모델 저장
import joblib

joblib.dump(model, "saved_models/xgb_model.pkl")
4.10 예측 실행
model = joblib.load("saved_models/xgb_model.pkl")

pred = model.predict(X)

feature_df["predicted_score"] = pred
4.11 위험등급 변환
feature_df["predicted_level"] = feature_df["predicted_score"].apply(classify_risk)
4.12 결과 저장
feature_df.to_csv("data/processed/result.csv", index=False)

import shap

6. 규칙 기반 모델 (보조)
목적
데이터 부족 시 fallback
def rule_based_score(row):
    score = 0
    score += row["past_sinkhole_count"] * 2
    score += row["gpr_detected_count"] * 3
    score += row["rainfall_7d"] * 0.5
    return score

규칙 기반 모델은 반드시 유지
